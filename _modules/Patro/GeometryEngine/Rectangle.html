    @property
    def p01(self):
        return self.__vector_cls__(self._p0.x, self._p1.y)

    @property
    def p10(self):
        return self.__vector_cls__(self._p1.x, self._p0.y)

    @property
    def edges(self):

        p0 = self._p0
        p1 = self.p01
        p2 = self._p1
        p3 = self.p10

        return (
            Segment2D(p0, p1),
            Segment2D(p1, p2),
            Segment2D(p2, p3),
            Segment2D(p3, p0),
        )

    ##############################################

    @property
    def diagonal(self):
        return self._p1 - self._p0

    ##############################################

    @property
    def perimeter(self):
        d = self.diagonal
        return 2*(abs(d.x) + abs(d.y))

    ##############################################

    @property
    def area(self):
        d = self.diagonal
        return abs(d.x * d.y)
